```go
package ado

import (
    "fmt"
    "net/url"
)

// WikiInfo represents an ADO wiki.
type WikiInfo struct {
    ID   string `json:"id"`
    Name string `json:"name"`
    Type string `json:"type"` // "projectWiki" or "codeWiki"
    URL  string `json:"url"`
}

// WikiPageInfo represents a page in an ADO wiki tree.
type WikiPageInfo struct {
    ID          int            `json:"id"`
    Path        string         `json:"path"`
    Order       int            `json:"order"`
    IsParent    bool           `json:"isParentPage"`
    GitItemPath string         `json:"gitItemPath"`
    Content     string         `json:"content"`
    SubPages    []WikiPageInfo `json:"subPages"`
    URL         string         `json:"url"`
}

// ListWikis fetches all wikis in the configured project.
// ADO API: GET {org}/{project}/_apis/wiki/wikis?api-version=7.1
func ListWikis(c *Client) ([]WikiInfo, error) {
    u := c.apiURL("wiki/wikis?api-version=7.1")
    resp, err := c.doRequest("GET", u, nil, "")
    if err != nil {
        return nil, fmt.Errorf("list wikis: %w", err)
    }
    var result struct {
        Value []WikiInfo `json:"value"`
    }
    if err := decodeResponse(resp, &result); err != nil {
        return nil, fmt.Errorf("parse wikis response: %w", err)
    }
    return result.Value, nil
}

// GetWikiPageTree fetches the full page tree for a wiki (recursive).
// ADO API: GET {org}/{project}/_apis/wiki/wikis/{wikiId}/pages?path=/&recursionLevel=full&api-version=7.1
func GetWikiPageTree(c *Client, wikiID string) (*WikiPageInfo, error) {
    u := c.apiURL(fmt.Sprintf("wiki/wikis/%s/pages?path=/&recursionLevel=full&api-version=7.1",
        url.PathEscape(wikiID)))
    resp, err := c.doRequest("GET", u, nil, "")
    if err != nil {
        return nil, fmt.Errorf("get wiki page tree: %w", err)
    }
    var page WikiPageInfo
    if err := decodeResponse(resp, &page); err != nil {
        return nil, fmt.Errorf("parse wiki page tree: %w", err)
    }
    return &page, nil
}

// GetWikiPageContent fetches a specific wiki page with content.
// ADO API: GET {org}/{project}/_apis/wiki/wikis/{wikiId}/pages?path={path}&includeContent=true&api-version=7.1
func GetWikiPageContent(c *Client, wikiID, pagePath string) (*WikiPageInfo, error) {
    u := c.apiURL(fmt.Sprintf("wiki/wikis/%s/pages?path=%s&includeContent=true&api-version=7.1",
        url.PathEscape(wikiID), url.QueryEscape(pagePath)))
    resp, err := c.doRequest("GET", u, nil, "")
    if err != nil {
        return nil, fmt.Errorf("get wiki page content: %w", err)
    }
    var page WikiPageInfo
    if err := decodeResponse(resp, &page); err != nil {
        return nil, fmt.Errorf("parse wiki page content: %w", err)
    }
    return &page, nil
}
```

package app

import (
    "fmt"
    "log"
    "path"
    "strings"

    "dev.azure.com/xbox/xb-tasks/domain"
    "dev.azure.com/xbox/xb-tasks/internal/config"
    "dev.azure.com/xbox/xb-tasks/internal/db"
    "dev.azure.com/xbox/xb-tasks/pkg/ado"

    "github.com/wailsapp/wails/v3/pkg/application"
)

// ADOWikiService handles browsing and importing ADO wiki pages.
type ADOWikiService struct {
    db        *db.DB
    tokenProv interface{ GetToken() (string, error) }
    cfg       *config.ConfigService
    app       *application.App
}

// NewADOWikiService creates an ADOWikiService.
func NewADOWikiService(database *db.DB, tokenProv interface{ GetToken() (string, error) }, cfg *config.ConfigService, wailsApp *application.App) *ADOWikiService {
    return &ADOWikiService{db: database, tokenProv: tokenProv, cfg: cfg, app: wailsApp}
}

// ADOWikiEntry represents a wiki visible in the browser (with org/project context).
type ADOWikiEntry struct {
    ID      string `json:"id"`
    Name    string `json:"name"`
    Type    string `json:"type"`
    Org     string `json:"org"`
    Project string `json:"project"`
}

// ListADOWikis fetches wikis from all configured ADO orgs.
func (s *ADOWikiService) ListADOWikis() ([]ADOWikiEntry, error) {
    token, err := s.tokenProv.GetToken()
    if err != nil {
        return nil, fmt.Errorf("get token: %w", err)
    }

    orgs := s.cfg.GetADOOrgs()
    var wikis []ADOWikiEntry
    for _, op := range orgs {
        client := ado.NewClient(op.Org, op.Project, token)
        wikiList, err := ado.ListWikis(client)
        if err != nil {
            log.Printf("[ADOWiki] Failed to list wikis for %s/%s: %v", op.Org, op.Project, err)
            continue
        }
        for _, w := range wikiList {
            wikis = append(wikis, ADOWikiEntry{
                ID: w.ID, Name: w.Name, Type: w.Type,
                Org: op.Org, Project: op.Project,
            })
        }
    }
    return wikis, nil
}

// GetADOWikiPageTree fetches the page tree for a specific wiki.
func (s *ADOWikiService) GetADOWikiPageTree(org, project, wikiID string) (*ado.WikiPageInfo, error) {
    token, err := s.tokenProv.GetToken()
    if err != nil {
        return nil, fmt.Errorf("get token: %w", err)
    }
    client := ado.NewClient(org, project, token)
    return ado.GetWikiPageTree(client, wikiID)
}

// ImportADOWikiPages imports selected pages from an ADO wiki as read-only mirrors.
// Runs content fetching in a background goroutine, emits progress events.
func (s *ADOWikiService) ImportADOWikiPages(org, project, wikiID string, pagePaths []string) error {
    go func() {
        token, err := s.tokenProv.GetToken()
        if err != nil {
            log.Printf("[ADOWiki] Token error during import: %v", err)
            s.app.EmitEvent("wiki:import:failed", map[string]string{"error": err.Error()})
            return
        }
        client := ado.NewClient(org, project, token)
        imported := 0
        for i, pp := range pagePaths {
            pageContent, err := ado.GetWikiPageContent(client, wikiID, pp)
            if err != nil {
                log.Printf("[ADOWiki] Failed to fetch content for %s: %v", pp, err)
                continue
            }
            // Use the last segment of the path as the title (strip leading /)
            title := path.Base(strings.ReplaceAll(pp, "\\", "/"))
            if title == "." || title == "/" {
                title = pp
            }

            already, _ := s.db.IsADOWikiPageImported(wikiID, pp)
            if already {
                // Update existing import
                // Find by ado_wiki_id + ado_page_path, update content
                pages, _ := s.db.ListADOWikiPages()
                for _, ep := range pages {
                    if ep.AdoWikiID == wikiID && ep.AdoPagePath == pp {
                        s.db.UpdateWikiPageContent(ep.ID, pageContent.Content)
                        break
                    }
                }
            } else {
                _, err = s.db.CreateADOWikiPage(title, pageContent.Content, wikiID, pp, org, project)
                if err != nil {
                    log.Printf("[ADOWiki] Failed to save page %s: %v", pp, err)
                    continue
                }
            }
            imported++
            s.app.EmitEvent("wiki:import:progress", map[string]interface{}{
                "current": i + 1, "total": len(pagePaths),
            })
        }
        s.app.EmitEvent("wiki:import:complete", map[string]interface{}{
            "imported": imported, "total": len(pagePaths),
        })
    }()
    return nil
}

// RefreshImportedPages re-fetches content for all ADO-sourced wiki pages.
func (s *ADOWikiService) RefreshImportedPages() error {
    pages, err := s.db.ListADOWikiPages()
    if err != nil {
        return fmt.Errorf("list ado wiki pages: %w", err)
    }
    if len(pages) == 0 {
        return nil
    }

    token, err := s.tokenProv.GetToken()
    if err != nil {
        return fmt.Errorf("get token for refresh: %w", err)
    }

    for _, page := range pages {
        client := ado.NewClient(page.AdoOrg, page.AdoProject, token)
        content, err := ado.GetWikiPageContent(client, page.AdoWikiID, page.AdoPagePath)
        if err != nil {
            log.Printf("[ADOWiki] Failed to refresh %s: %v", page.AdoPagePath, err)
            continue
        }
        s.db.UpdateWikiPageContent(page.ID, content.Content)
    }
    return nil
}

package app

import (
    "dev.azure.com/xbox/xb-tasks/domain"
    "dev.azure.com/xbox/xb-tasks/internal/db"
)

// WikiService manages local wiki pages. Methods are exposed as Wails bindings.
type WikiService struct {
    db *db.DB
}

// NewWikiService creates a WikiService.
func NewWikiService(database *db.DB) *WikiService {
    return &WikiService{db: database}
}

// CreatePage creates a new wiki page with the given title and content.
func (s *WikiService) CreatePage(title, content string) (domain.WikiPage, error) {
    return s.db.CreateWikiPage(title, content)
}

// GetPage returns a wiki page by ID.
func (s *WikiService) GetPage(id int) (domain.WikiPage, error) {
    return s.db.GetWikiPage(id)
}

// ListPages returns all wiki pages sorted by title.
func (s *WikiService) ListPages() ([]domain.WikiPage, error) {
    return s.db.ListWikiPages()
}

// UpdatePage updates a wiki page's title and content. Returns error for read-only pages.
func (s *WikiService) UpdatePage(id int, title, content string) (domain.WikiPage, error) {
    return s.db.UpdateWikiPage(id, title, content)
}

// DeletePage removes a wiki page by ID.
func (s *WikiService) DeletePage(id int) error {
    return s.db.DeleteWikiPage(id)
}

// Search performs FTS5 full-text search across all wiki pages.
