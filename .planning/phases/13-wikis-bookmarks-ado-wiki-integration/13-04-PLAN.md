The view has two panels side by side. Left: WikiSidebar (280px fixed). Right: WikiPageHeader + WikiEditor/WikiPreview (flex-1).

Structure:
```
<script setup lang="ts">
import { onMounted, watch } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { useMagicKeys } from '@vueuse/core'
import { useWikiStore } from '@/stores/wiki'
import WikiSidebar from '@/components/wiki/WikiSidebar.vue'
import WikiPageHeader from '@/components/wiki/WikiPageHeader.vue'
import WikiEditor from '@/components/wiki/WikiEditor.vue'
import WikiPreview from '@/components/wiki/WikiPreview.vue'
import EmptyState from '@/components/EmptyState.vue'
import { BookOpen } from 'lucide-vue-next'
import { Button } from '@/components/ui/button'
import { ScrollArea } from '@/components/ui/scroll-area'

const route = useRoute()
const router = useRouter()
const wikiStore = useWikiStore()

// Keyboard shortcuts (matching TasksView pattern)
const keys = useMagicKeys()
watch(keys['Meta+n'], v => { if (v) handleNewPage() })
watch(keys['Ctrl+n'], v => { if (v) handleNewPage() })
watch(keys['Meta+e'], v => { if (v) wikiStore.toggleEditMode() })
watch(keys['Ctrl+e'], v => { if (v) wikiStore.toggleEditMode() })

onMounted(async () => {
  wikiStore.initEvents()
  await wikiStore.fetchPages()
  // If route has :id param, select that page
  const id = Number(route.params.id)
  if (id) wikiStore.selectPage(id)
})

// Sync route param with selected page
watch(() => route.params.id, (newId) => {
  if (newId) wikiStore.selectPage(Number(newId))
})

async function handleNewPage() {
  const page = await wikiStore.createPage('Untitled')
  if (page) router.replace({ name: 'wiki-page', params: { id: page.id } })
}
</script>
```

- **Wiki link click handling**: Add a `@click` handler on the content div that checks for `a[data-wiki-page]` clicks:
  ```typescript
  function handleClick(event: MouseEvent) {
    const target = event.target as HTMLElement
    const link = target.closest('a[data-wiki-page]')
    if (link) {
      event.preventDefault()
      const pageName = link.getAttribute('data-wiki-page')
      if (!pageName) return
      const existing = wikiStore.pages.find(p => p.title === pageName)
      if (existing) {
        wikiStore.selectPage(existing.id)
        router.push({ name: 'wiki-page', params: { id: existing.id } })
      } else {
        // Create-on-click for missing pages
        wikiStore.createPage(pageName).then(page => {
          if (page) router.push({ name: 'wiki-page', params: { id: page.id } })
        })
      }
    }
  }
  ```
