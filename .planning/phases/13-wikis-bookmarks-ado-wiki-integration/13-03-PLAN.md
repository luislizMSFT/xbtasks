**Create `frontend/src/composables/useMarkdown.ts`:**

```typescript
import { marked } from 'marked'
import DOMPurify from 'dompurify'

/**
 * Pre-process [[Page Name]] wiki links before marked parses markdown.
 * Converts to HTML anchors with data-wiki-page attribute for click handling.
 * Existing pages get class "wiki-link", missing pages get "wiki-link wiki-link-missing".
 */
function resolveWikiLinks(markdown: string, existingPages: Set<string>): string {
  return markdown.replace(/\[\[([^\]]+)\]\]/g, (_match, pageName: string) => {
    const trimmed = pageName.trim()
    const exists = existingPages.has(trimmed)
    const cssClass = exists ? 'wiki-link' : 'wiki-link wiki-link-missing'
    return `<a href="#" class="${cssClass}" data-wiki-page="${trimmed}">${trimmed}</a>`
  })
}

const ALLOWED_TAGS = [
  'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
  'p', 'a', 'strong', 'em', 'code', 'pre',
  'ul', 'ol', 'li', 'blockquote', 'img',
  'mark', 'br', 'hr',
  'table', 'thead', 'tbody', 'tr', 'th', 'td',
  'del', 'input', // for task lists
]

const ALLOWED_ATTR = ['href', 'src', 'alt', 'class', 'data-wiki-page', 'type', 'checked', 'disabled']

/**
 * Composable for markdown rendering with wiki link support.
 * Uses marked for parsing, DOMPurify for XSS-safe sanitization.
 */
export function useMarkdown() {
  /**
   * Render markdown to sanitized HTML with wiki link resolution.
   * @param content Raw markdown string
   * @param existingPages Set of page titles that exist (for link styling)
   */
  function render(content: string, existingPages: Set<string>): string {
    // Step 1: Pre-process wiki links BEFORE marked (pitfall 2 — must be before marked)
    const withLinks = resolveWikiLinks(content, existingPages)
    // Step 2: Parse markdown to HTML
    const html = marked.parse(withLinks, { async: false }) as string
    // Step 3: Sanitize to prevent XSS
    return DOMPurify.sanitize(html, {
      ALLOWED_TAGS,
      ALLOWED_ATTR,
    })
  }

  /**
   * Render a plain search snippet (already contains <mark> from FTS5) — sanitize only.
   */
  function renderSnippet(snippet: string): string {
    return DOMPurify.sanitize(snippet, {
      ALLOWED_TAGS: ['mark'],
      ALLOWED_ATTR: [],
    })
  }

  return { render, renderSnippet }
}
```

**Create `frontend/src/stores/wiki.ts`:**

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { useDebounceFn } from '@vueuse/core'
import type { WikiPage, WikiSearchResult } from '@/types'
import * as wikiApi from '@/api/wiki'
import { toast } from 'vue-sonner'
import { Events } from '@wailsio/runtime'

export const useWikiStore = defineStore('wiki', () => {
  // --- State ---
  const pages = ref<WikiPage[]>([])
  const selectedPageId = ref<number | null>(null)
  const loading = ref(false)
  const editMode = ref(false)
  const searchQuery = ref('')
  const searchResults = ref<WikiSearchResult[]>([])
  const isSearching = ref(false)

  // Flight guard (prevents duplicate fetches)
  let fetchInFlight = false

  // --- Computed ---
  const selectedPage = computed<WikiPage | null>(() =>
    pages.value.find(p => p.id === selectedPageId.value) ?? null
  )

  const sortedPages = computed(() =>
    [...pages.value].sort((a, b) => a.title.localeCompare(b.title))
  )

  const pageTitles = computed(() => new Set(pages.value.map(p => p.title)))

  const localPages = computed(() => pages.value.filter(p => p.source === 'local'))
  const adoPages = computed(() => pages.value.filter(p => p.source === 'ado'))

  // --- Debounced search (200ms, matching ADO tree search pattern) ---
  const doSearch = useDebounceFn(async (q: string) => {
    if (!q.trim()) {
      searchResults.value = []
      isSearching.value = false
      return
    }
    isSearching.value = true
    try {
      searchResults.value = (await wikiApi.search(q)) as WikiSearchResult[]
    } catch {
      searchResults.value = []
    } finally {
      isSearching.value = false
    }
  }, 200)

  // --- Actions ---
  async function fetchPages() {
    if (fetchInFlight) return
    fetchInFlight = true
    loading.value = true
    try {
      pages.value = (await wikiApi.listPages()) as WikiPage[]
    } catch {
      pages.value = []
    } finally {
      loading.value = false
      fetchInFlight = false
    }
  }

  function selectPage(id: number) {
    selectedPageId.value = id
  }

  async function createPage(title: string, content: string = '') {
    try {
      const page = (await wikiApi.createPage(title, content)) as WikiPage
      pages.value.push(page)
      selectedPageId.value = page.id
      editMode.value = true
      return page
    } catch (e) {
      toast.error('Failed to create page')
      return null
    }
  }

  async function updatePage(id: number, title: string, content: string) {
    try {
      const updated = (await wikiApi.updatePage(id, title, content)) as WikiPage
      const idx = pages.value.findIndex(p => p.id === id)
      if (idx !== -1) pages.value[idx] = updated
      return updated
    } catch (e) {
      toast.error('Failed to save page. Your changes are preserved — try again.')
      return null
    }
  }

  async function deletePage(id: number) {
    try {
      await wikiApi.deletePage(id)
      pages.value = pages.value.filter(p => p.id !== id)
      if (selectedPageId.value === id) {
        selectedPageId.value = null
        editMode.value = false
      }
    } catch (e) {
      toast.error('Failed to delete page')
    }
  }

  function toggleEditMode() {
    editMode.value = !editMode.value
  }

  function clearSearch() {
    searchQuery.value = ''
    searchResults.value = []
  }

  // --- Wails event listeners for ADO wiki import ---
  function initEvents() {
    Events.On('wiki:import:complete', (data: any) => {
      const d = data?.data?.[0] ?? data
      toast.success(`Imported ${d.imported} pages`)
      fetchPages() // Refresh page list after import
    })
    Events.On('wiki:import:failed', (data: any) => {
      const d = data?.data?.[0] ?? data
      toast.error(`Wiki import failed: ${d.error}`)
    })
  }

  return {
    // State
    pages, selectedPageId, loading, editMode, searchQuery, searchResults, isSearching,
    // Computed
    selectedPage, sortedPages, pageTitles, localPages, adoPages,
    // Actions
    fetchPages, selectPage, createPage, updatePage, deletePage,
    toggleEditMode, doSearch, clearSearch, initEvents,
  }
})
```
