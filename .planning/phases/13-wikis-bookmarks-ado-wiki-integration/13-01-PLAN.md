```go
package domain

import "time"

// WikiPage represents a local or ADO-imported wiki page.
type WikiPage struct {
    ID          int       `json:"id"`
    Title       string    `json:"title"`
    Content     string    `json:"content"`
    Source      string    `json:"source"`      // "local" or "ado"
    IsReadonly  bool      `json:"isReadonly"`
    AdoWikiID   string    `json:"adoWikiId"`
    AdoPagePath string    `json:"adoPagePath"`
    AdoOrg      string    `json:"adoOrg"`
    AdoProject  string    `json:"adoProject"`
    CreatedAt   time.Time `json:"createdAt"`
    UpdatedAt   time.Time `json:"updatedAt"`
}

// WikiSearchResult represents an FTS5 search hit.
type WikiSearchResult struct {
    ID         int    `json:"id"`
    Title      string `json:"title"`
    Source     string `json:"source"`
    Preview    string `json:"preview"`   // snippet from FTS5
    IsReadonly bool   `json:"isReadonly"`
}

// ProjectBookmark represents an external URL bookmarked on a project.
type ProjectBookmark struct {
    ID        int       `json:"id"`
    ProjectID int       `json:"projectId"`
    URL       string    `json:"url"`
    Label     string    `json:"label"`
    Type      string    `json:"type"` // url, icm, grafana, ado, wiki
    CreatedAt time.Time `json:"createdAt"`
}
```

```sql
-- ─── Wiki ───

CREATE TABLE IF NOT EXISTS wiki_pages (
    id            INTEGER PRIMARY KEY AUTOINCREMENT,
    title         TEXT NOT NULL,
    content       TEXT DEFAULT '',
    source        TEXT DEFAULT 'local' CHECK(source IN ('local','ado')),
    is_readonly   INTEGER DEFAULT 0,
    ado_wiki_id   TEXT DEFAULT '',
    ado_page_path TEXT DEFAULT '',
    ado_org       TEXT DEFAULT '',
    ado_project   TEXT DEFAULT '',
    created_at    DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at    DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE VIRTUAL TABLE IF NOT EXISTS wiki_fts USING fts5(
    title, content,
    content=wiki_pages, content_rowid=id,
    tokenize='unicode61 remove_diacritics 2'
);

CREATE TRIGGER IF NOT EXISTS wiki_fts_insert AFTER INSERT ON wiki_pages BEGIN
    INSERT INTO wiki_fts(rowid, title, content)
    VALUES (new.id, new.title, new.content);
END;

CREATE TRIGGER IF NOT EXISTS wiki_fts_delete BEFORE DELETE ON wiki_pages BEGIN
    INSERT INTO wiki_fts(wiki_fts, rowid, title, content)
    VALUES ('delete', old.id, old.title, old.content);
END;

CREATE TRIGGER IF NOT EXISTS wiki_fts_update_before BEFORE UPDATE ON wiki_pages BEGIN
    INSERT INTO wiki_fts(wiki_fts, rowid, title, content)
    VALUES ('delete', old.id, old.title, old.content);
END;

CREATE TRIGGER IF NOT EXISTS wiki_fts_update_after AFTER UPDATE ON wiki_pages BEGIN
    INSERT INTO wiki_fts(rowid, title, content)
    VALUES (new.id, new.title, new.content);
END;

-- ─── Project bookmarks ───

CREATE TABLE IF NOT EXISTS project_bookmarks (
    id         INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    url        TEXT NOT NULL,
    label      TEXT DEFAULT '',
    type       TEXT DEFAULT 'url',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

```go
package db

import (
    "fmt"
    "strings"
    "time"

    "dev.azure.com/xbox/xb-tasks/domain"
)

// CreateWikiPage inserts a new local wiki page. FTS5 trigger auto-updates index.
func (db *DB) CreateWikiPage(title, content string) (domain.WikiPage, error) {
    res, err := db.Exec(
        `INSERT INTO wiki_pages (title, content) VALUES (?, ?)`,
        title, content,
    )
    if err != nil {
        return domain.WikiPage{}, fmt.Errorf("create wiki page: %w", err)
    }
    id, _ := res.LastInsertId()
    return db.GetWikiPage(int(id))
}

// GetWikiPage returns a single wiki page by ID.
func (db *DB) GetWikiPage(id int) (domain.WikiPage, error) {
    var p domain.WikiPage
    var isReadonly int
    err := db.QueryRow(`
        SELECT id, title, content, source, is_readonly,
               ado_wiki_id, ado_page_path, ado_org, ado_project,
               created_at, updated_at
        FROM wiki_pages WHERE id = ?`, id,
    ).Scan(&p.ID, &p.Title, &p.Content, &p.Source, &isReadonly,
        &p.AdoWikiID, &p.AdoPagePath, &p.AdoOrg, &p.AdoProject,
        &p.CreatedAt, &p.UpdatedAt)
    p.IsReadonly = isReadonly != 0
    return p, err
}

// ListWikiPages returns all wiki pages sorted by title.
func (db *DB) ListWikiPages() ([]domain.WikiPage, error) {
    rows, err := db.Query(`
        SELECT id, title, content, source, is_readonly,
               ado_wiki_id, ado_page_path, ado_org, ado_project,
               created_at, updated_at
        FROM wiki_pages ORDER BY title COLLATE NOCASE`)
    if err != nil {
        return nil, fmt.Errorf("list wiki pages: %w", err)
    }
    defer rows.Close()

    var pages []domain.WikiPage
    for rows.Next() {
        var p domain.WikiPage
        var isReadonly int
        if err := rows.Scan(&p.ID, &p.Title, &p.Content, &p.Source, &isReadonly,
            &p.AdoWikiID, &p.AdoPagePath, &p.AdoOrg, &p.AdoProject,
            &p.CreatedAt, &p.UpdatedAt); err != nil {
            return nil, err
        }
        p.IsReadonly = isReadonly != 0
        pages = append(pages, p)
    }
    return pages, rows.Err()
}

// UpdateWikiPage updates title and content. Returns error if page is read-only.
func (db *DB) UpdateWikiPage(id int, title, content string) (domain.WikiPage, error) {
    // Enforce read-only at DB level
    var isReadonly int
    if err := db.QueryRow(`SELECT is_readonly FROM wiki_pages WHERE id = ?`, id).Scan(&isReadonly); err != nil {
        return domain.WikiPage{}, fmt.Errorf("check wiki page: %w", err)
    }
    if isReadonly != 0 {
        return domain.WikiPage{}, fmt.Errorf("cannot update read-only wiki page %d", id)
    }

    _, err := db.Exec(`
        UPDATE wiki_pages SET title = ?, content = ?, updated_at = ?
        WHERE id = ?`, title, content, time.Now(), id)
    if err != nil {
        return domain.WikiPage{}, fmt.Errorf("update wiki page: %w", err)
    }
    return db.GetWikiPage(id)
}

// UpdateWikiPageContent updates only the content of a page (used for ADO re-fetch).
func (db *DB) UpdateWikiPageContent(id int, content string) error {
    _, err := db.Exec(`
        UPDATE wiki_pages SET content = ?, updated_at = ?
        WHERE id = ?`, content, time.Now(), id)
    return err
}

// DeleteWikiPage removes a wiki page by ID. FTS5 trigger auto-cleans index.
func (db *DB) DeleteWikiPage(id int) error {
    _, err := db.Exec(`DELETE FROM wiki_pages WHERE id = ?`, id)
    return err
}

// CreateADOWikiPage inserts an ADO-sourced read-only wiki page.
func (db *DB) CreateADOWikiPage(title, content, adoWikiID, adoPagePath, adoOrg, adoProject string) (domain.WikiPage, error) {
    res, err := db.Exec(`
        INSERT INTO wiki_pages (title, content, source, is_readonly, ado_wiki_id, ado_page_path, ado_org, ado_project)
        VALUES (?, ?, 'ado', 1, ?, ?, ?, ?)`,
        title, content, adoWikiID, adoPagePath, adoOrg, adoProject)
    if err != nil {
        return domain.WikiPage{}, fmt.Errorf("create ado wiki page: %w", err)
    }
    id, _ := res.LastInsertId()
    return db.GetWikiPage(int(id))
}

// ListADOWikiPages returns all ADO-sourced wiki pages (for re-fetch).
func (db *DB) ListADOWikiPages() ([]domain.WikiPage, error) {
    rows, err := db.Query(`
        SELECT id, title, content, source, is_readonly,
               ado_wiki_id, ado_page_path, ado_org, ado_project,
               created_at, updated_at
        FROM wiki_pages WHERE source = 'ado' ORDER BY title COLLATE NOCASE`)
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    var pages []domain.WikiPage
    for rows.Next() {
        var p domain.WikiPage
        var isReadonly int
        if err := rows.Scan(&p.ID, &p.Title, &p.Content, &p.Source, &isReadonly,
            &p.AdoWikiID, &p.AdoPagePath, &p.AdoOrg, &p.AdoProject,
            &p.CreatedAt, &p.UpdatedAt); err != nil {
            return nil, err
        }
        p.IsReadonly = isReadonly != 0
        pages = append(pages, p)
    }
    return pages, rows.Err()
}

// IsADOWikiPageImported checks if a specific ADO wiki page path is already imported.
func (db *DB) IsADOWikiPageImported(adoWikiID, adoPagePath string) (bool, error) {
    var count int
    err := db.QueryRow(`
        SELECT COUNT(*) FROM wiki_pages
        WHERE ado_wiki_id = ? AND ado_page_path = ?`, adoWikiID, adoPagePath).Scan(&count)
    return count > 0, err
}

// SearchWikiPages performs FTS5 search. Input is sanitized (wrapped in double quotes).
// Returns up to 20 results with snippet previews.
func (db *DB) SearchWikiPages(query string) ([]domain.WikiSearchResult, error) {
    if strings.TrimSpace(query) == "" {
        return nil, nil
    }
    // Sanitize: wrap in double quotes to prevent FTS5 syntax errors from user input
    safeQuery := `"` + strings.ReplaceAll(query, `"`, `""`) + `"`

    rows, err := db.Query(`
        SELECT wp.id, wp.title, wp.source, wp.is_readonly,
               snippet(wiki_fts, 1, '<mark>', '</mark>', '...', 32) as preview
        FROM wiki_fts
        JOIN wiki_pages wp ON wp.id = wiki_fts.rowid
        WHERE wiki_fts MATCH ?
        ORDER BY rank
        LIMIT 20`, safeQuery)
    if err != nil {
        return nil, fmt.Errorf("search wiki pages: %w", err)
    }
    defer rows.Close()

    var results []domain.WikiSearchResult
    for rows.Next() {
        var r domain.WikiSearchResult
        var isReadonly int
        if err := rows.Scan(&r.ID, &r.Title, &r.Source, &isReadonly, &r.Preview); err != nil {
            return nil, err
        }
        r.IsReadonly = isReadonly != 0
        results = append(results, r)
    }
    return results, rows.Err()
}
```

```go
// ListBookmarks returns all bookmarks for a project.
func (db *DB) ListBookmarks(projectID int) ([]domain.ProjectBookmark, error) {

        SELECT id, project_id, url, label, type, created_at
        FROM project_bookmarks
        WHERE project_id = ?

    return bookmarks, rows.Err()
```
