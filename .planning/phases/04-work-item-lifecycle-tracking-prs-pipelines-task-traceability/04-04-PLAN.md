**Layout computation with dagre:**

```typescript
import dagre from '@dagrejs/dagre'

const NODE_WIDTH = 180   // wider than UI-SPEC's 48px to fit text
const NODE_HEIGHT = 40   // taller to fit icon + title

function computeLayout(nodes: ForceGraphNode[], edges: ForceGraphEdge[]) {
  // Handle disconnected components: group by connectivity, layout each, arrange in grid
  const components = findConnectedComponents(nodes, edges)
  const allPositioned: { node: ForceGraphNode; x: number; y: number }[] = []
  let offsetX = 0

  for (const comp of components) {
    const g = new dagre.graphlib.Graph()
    g.setGraph({
      rankdir: 'TB',    // Top-to-bottom per D-13
      nodesep: 24,       // Horizontal spacing
      ranksep: 64,       // Vertical spacing between ranks
      edgesep: 16,
      marginx: 20,
      marginy: 20,
    })
    g.setDefaultEdgeLabel(() => ({}))

    for (const node of comp.nodes) {
      g.setNode(String(node.id), { width: NODE_WIDTH, height: NODE_HEIGHT })
    }
    for (const edge of comp.edges) {
      g.setEdge(String(edge.source), String(edge.target))
    }

    dagre.layout(g)

    const graphInfo = g.graph()
    for (const node of comp.nodes) {
      const pos = g.node(String(node.id))
      allPositioned.push({ node, x: pos.x + offsetX, y: pos.y })
    }
    offsetX += (graphInfo.width || 200) + 60  // gap between disconnected components
  }

  return allPositioned
}

// Find connected components for separate dagre layouts (pitfall 3)
function findConnectedComponents(nodes: ForceGraphNode[], edges: ForceGraphEdge[]) {
  const adj = new Map<number, Set<number>>()
  for (const n of nodes) adj.set(n.id, new Set())
  for (const e of edges) {
    adj.get(e.source)?.add(e.target)
    adj.get(e.target)?.add(e.source)
  }

  const visited = new Set<number>()
  const components: { nodes: ForceGraphNode[]; edges: ForceGraphEdge[] }[] = []

  for (const node of nodes) {
    if (visited.has(node.id)) continue
    const compNodes: ForceGraphNode[] = []
    const queue = [node.id]
    while (queue.length) {
      const id = queue.shift()!
      if (visited.has(id)) continue
      visited.add(id)
      compNodes.push(nodes.find(n => n.id === id)!)
      for (const neighbor of adj.get(id) || []) {
        if (!visited.has(neighbor)) queue.push(neighbor)
      }
    }
    const nodeIds = new Set(compNodes.map(n => n.id))
    const compEdges = edges.filter(e => nodeIds.has(e.source) && nodeIds.has(e.target))
    components.push({ nodes: compNodes, edges: compEdges })
  }

  return components
}
```

**Build graph data from real stores (D-16):**

```typescript
const graphNodes = computed<ForceGraphNode[]>(() => {
  const nodes: ForceGraphNode[] = []

  // Task nodes
  for (const task of taskStore.tasks) {
    nodes.push({
      id: task.id,
      title: task.title,
      status: task.status,
      priority: task.priority,
      projectId: task.projectId,
      source: 'local',
      entityType: 'task',
      adoId: task.adoId || undefined,
    })
  }

  // ADO work item nodes (from ado store cache)
  for (const wi of adoStore.workItems) {
    // Use negative IDs to avoid collision with task IDs
    nodes.push({
      id: -wi.id,  // negative to avoid collision
      title: wi.title,
      status: wi.state,
      priority: `P${wi.priority}`,
      projectId: null,
      source: 'ado',
      entityType: 'ado-work-item',
      adoId: wi.adoId,
      adoUrl: wi.url,
      type: wi.type,
    })
  }

  // PR nodes (use large offset to avoid collision)
  for (const pr of [...prStore.myPRs, ...prStore.reviewPRs]) {
    const prNodeId = 100000 + pr.prNumber  // offset to avoid collision
    if (nodes.some(n => n.id === prNodeId)) continue // dedup
    nodes.push({
      id: prNodeId,
      title: pr.title,
      status: pr.status,
      priority: '',
      projectId: null,
      source: 'pr',
      entityType: 'pr',
    })
  }

  // Pipeline nodes (use different offset)
  for (const pipe of adoStore.pipelines) {
    nodes.push({
      id: 200000 + pipe.id,  // offset
      title: pipe.name,
      status: pipe.result || pipe.status,
      priority: '',
      projectId: null,
      source: 'pipeline',
      entityType: 'pipeline',
    })
  }

  return nodes
})

const graphEdges = computed<ForceGraphEdge[]>(() => {
  const edges: ForceGraphEdge[] = []

  // Task dependency edges
  for (const task of taskStore.tasks) {
    // Parent-child edges
    if (task.parentId) {
      edges.push({ source: task.parentId, target: task.id, kind: 'parent' })
    }
    // Task → ADO work item link
    if (task.adoId) {
      const adoNode = graphNodes.value.find(
        n => n.entityType === 'ado-work-item' && n.adoId === task.adoId
      )
      if (adoNode) {
        edges.push({ source: adoNode.id, target: task.id, kind: 'ado-link' })
      }
    }
  }

  // ADO parent-child hierarchy
  for (const wi of adoStore.workItems) {
    if (wi.parentId > 0) {
      const parentNode = graphNodes.value.find(
        n => n.entityType === 'ado-work-item' && n.adoId === String(wi.parentId)
      )
      if (parentNode) {
        edges.push({ source: parentNode.id, target: -wi.id, kind: 'parent' })
      }
    }
  }

  // PR → Task edges (via taskId on PR)
  for (const pr of [...prStore.myPRs, ...prStore.reviewPRs]) {
    const prNodeId = 100000 + pr.prNumber
    if (pr.taskId) {
      edges.push({ source: pr.taskId, target: prNodeId, kind: 'pr-link' })
    }
    // PR → ADO work item (via adoId on PR)
    if (pr.adoId) {
      const adoNode = graphNodes.value.find(
        n => n.entityType === 'ado-work-item' && n.adoId === pr.adoId
      )
      if (adoNode) {
        edges.push({ source: adoNode.id, target: prNodeId, kind: 'pr-link' })
      }
    }
  }

  // Pipeline → PR edges (branch name match per D-02)
  for (const pipe of adoStore.pipelines) {
    const pipeNodeId = 200000 + pipe.id
    for (const pr of [...prStore.myPRs, ...prStore.reviewPRs]) {
      const prBranch = pr.sourceBranch.replace('refs/heads/', '')
      if (pipe.sourceBranch === prBranch) {
        const prNodeId = 100000 + pr.prNumber
        edges.push({ source: prNodeId, target: pipeNodeId, kind: 'pipeline-link' })
      }
    }
  }

  // Filter edges to only those where both source and target exist in nodes
  const nodeIds = new Set(graphNodes.value.map(n => n.id))
  return edges.filter(e => nodeIds.has(e.source) && nodeIds.has(e.target))
})
```
