package ado

import "fmt"

// PRResult matches the ADO REST API response for a pull request.
type PRResult struct {
    PullRequestID int      `json:"pullRequestId"`
    Title         string   `json:"title"`
    Description   string   `json:"description"`
    Status        string   `json:"status"`
    IsDraft       bool     `json:"isDraft"`
    Repository    struct {
        Name string `json:"name"`
    } `json:"repository"`
    SourceRefName string   `json:"sourceRefName"`
    TargetRefName string   `json:"targetRefName"`
    CreatedBy     Identity `json:"createdBy"`
    Reviewers     []struct {
        DisplayName string `json:"displayName"`
        UniqueName  string `json:"uniqueName"`
        Vote        int    `json:"vote"`
    } `json:"reviewers"`
    CreationDate string  `json:"creationDate"`
    ClosedDate   *string `json:"closedDate"`
    URL          string  `json:"url"`
}

// ListPullRequests returns PRs matching the given criteria.
// creatorID is the user GUID from GetCurrentUserID.
// status: "active", "completed", "abandoned", "all".
func ListPullRequests(c *Client, creatorID string, status string, top int) ([]PRResult, error) {
    url := c.apiURL(fmt.Sprintf(
        "git/pullrequests?searchCriteria.creatorId=%s&searchCriteria.status=%s&$top=%d&api-version=7.0",
        creatorID, status, top,
    ))
    resp, err := c.doRequest("GET", url, nil, "")
    if err != nil {
        return nil, fmt.Errorf("list PRs: %w", err)
    }
    var result struct {
        Value []PRResult `json:"value"`
    }
    return result.Value, decodeResponse(resp, &result)
}

// ListReviewPullRequests returns active PRs where the given user is a reviewer.
func ListReviewPullRequests(c *Client, reviewerID string, top int) ([]PRResult, error) {
    url := c.apiURL(fmt.Sprintf(
        "git/pullrequests?searchCriteria.reviewerId=%s&searchCriteria.status=active&$top=%d&api-version=7.0",
        reviewerID, top,
    ))
    resp, err := c.doRequest("GET", url, nil, "")
    if err != nil {
        return nil, fmt.Errorf("list review PRs: %w", err)
    }
    var result struct {
        Value []PRResult `json:"value"`
    }
    return result.Value, decodeResponse(resp, &result)
}

// GetCurrentUserID returns the authenticated user's GUID from the Connection Data API.
// This GUID is required for PR queries (creatorId, reviewerId).
func GetCurrentUserID(c *Client) (string, error) {
    url := fmt.Sprintf("https://dev.azure.com/%s/_apis/connectionData", c.Org())
    resp, err := c.doRequest("GET", url, nil, "")
    if err != nil {
        return "", fmt.Errorf("get current user ID: %w", err)
    }
    var result struct {
        AuthenticatedUser struct {
            ID string `json:"id"`
        } `json:"authenticatedUser"`
    }
    if err := decodeResponse(resp, &result); err != nil {
        return "", fmt.Errorf("parse connection data: %w", err)
    }
    return result.AuthenticatedUser.ID, nil
}
```

package ado

import "fmt"

// BuildResult matches the ADO REST API response for a build/pipeline run.
type BuildResult struct {
    ID          int    `json:"id"`
    BuildNumber string `json:"buildNumber"`
    Status      string `json:"status"`  // notStarted, inProgress, completed
    Result      string `json:"result"`  // succeeded, failed, canceled, partiallySucceeded
    Definition  struct {
        Name string `json:"name"`
    } `json:"definition"`
    SourceBranch string  `json:"sourceBranch"`
    QueueTime    string  `json:"queueTime"`
    FinishTime   *string `json:"finishTime"`
    URL          string  `json:"url"`
}

// ListRecentBuilds returns the most recent build/pipeline runs for the project.
func ListRecentBuilds(c *Client, top int) ([]BuildResult, error) {
    url := c.apiURL(fmt.Sprintf("build/builds?$top=%d&api-version=7.0", top))
    resp, err := c.doRequest("GET", url, nil, "")
    if err != nil {
        return nil, fmt.Errorf("list builds: %w", err)
    }
    var result struct {
        Value []BuildResult `json:"value"`
    }
    return result.Value, decodeResponse(resp, &result)
}
```
