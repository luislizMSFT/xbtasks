```typescript
const snoozedPRs = ref<PRSnooze[]>([])

// Check if a PR is currently snoozed
function isSnoozed(prId: number, repo: string): boolean {
  return snoozedPRs.value.some(
    s => s.prId === prId && s.repo === repo && new Date(s.snoozeUntil) > new Date()
  )
}

// Get snooze expiry for a PR (null if not snoozed)
function getSnoozeUntil(prId: number, repo: string): string | null {
  const snooze = snoozedPRs.value.find(s => s.prId === prId && s.repo === repo)
  return snooze && new Date(snooze.snoozeUntil) > new Date() ? snooze.snoozeUntil : null
}

// Check if AB# suggestion was dismissed
function isABDismissed(prId: number, adoId: string): boolean {
  return abDismissals.value.some(d => d.prId === prId && d.adoId === adoId)
}

async function snoozePR(prId: number, repo: string, preset: string) {
```

**2. Add computeds for completed PRs and AB# detection:**

```typescript
// Completed PRs from last 7 days, not snoozed, max 5
const completedPRs = computed(() =>
  prStore.myPRs
    .filter(pr => pr.status === 'completed' && !prStore.isSnoozed(pr.prNumber, pr.repo))
    .filter(pr => {
      if (!pr.mergedAt) return false
      const mergedDate = new Date(pr.mergedAt)
      const weekAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)
      return mergedDate > weekAgo
    })
    .slice(0, 5)
)

// Track which PRs have been nudged this session (prevent repeat nudges per D-08)
const nudgedPRs = ref(new Set<string>())

// AB# detection: for each PR, detect refs and filter out dismissed ones
const abSuggestions = ref<Map<number, number[]>>(new Map())

// Detect AB# refs on data load
watch(() => [...prStore.myPRs, ...prStore.reviewPRs], async (prs) => {
  for (const pr of prs) {
    if (pr.status === 'abandoned') continue
    try {
      const refs = await prsApi.detectWorkItemRefs(pr.title, pr.description || '')
      const filtered = refs.filter((id: number) => !prStore.isABDismissed(pr.prNumber, String(id)))
      if (filtered.length > 0) {
        abSuggestions.value.set(pr.prNumber, filtered)
      }
    } catch { /* ignore detection failures */ }
  }
}, { immediate: true })
```

**3. Add merge nudge check (per D-07, D-08):**

Add a watch that detects when a previously-active linked PR becomes completed:

```typescript
// Track previous PR statuses to detect merge events
const prevPRStatuses = ref<Map<string, string>>(new Map())

watch(() => prStore.myPRs, (prs) => {
  for (const pr of prs) {
    const key = `${pr.prNumber}:${pr.repo}`
    const prev = prevPRStatuses.value.get(key)
    if (prev && prev !== 'completed' && pr.status === 'completed' && pr.taskId && !nudgedPRs.value.has(key)) {
      // PR just merged and has a linked task
      nudgedPRs.value.add(key)
      const linkedTask = taskStore.tasks.find(t => t.id === pr.taskId)
      if (linkedTask && linkedTask.status !== 'done') {
        toast.info('PR merged', {
          description: `${pr.title} was merged. Mark linked task as done?`,
          duration: 10000,
          action: {
            label: 'Mark Task Done',
            onClick: () => taskStore.setStatus(pr.taskId!, 'done'),
          },
          cancel: {
            label: 'Dismiss',
            onClick: () => {},
          },
        })
      }
    }
    prevPRStatuses.value.set(key, pr.status)
  }
})
```
