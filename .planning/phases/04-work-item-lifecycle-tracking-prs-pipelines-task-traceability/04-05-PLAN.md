**1. Time axis computation:**

```typescript
import { computed, ref, onMounted } from 'vue'
import { ScrollArea, ScrollBar } from '@/components/ui/scroll-area'
import type { TimelineEvent } from '@/types'

// Filter to last 30 days by default
const daysRange = ref(30)

const filteredEvents = computed(() => {
  const cutoff = new Date(Date.now() - daysRange.value * 24 * 60 * 60 * 1000)
  return props.events
    .filter(e => new Date(e.timestamp) >= cutoff)
    .sort((a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime())
})

// Compute time range
const timeRange = computed(() => {
  if (filteredEvents.value.length === 0) return { start: new Date(), end: new Date(), span: 1 }
  const times = filteredEvents.value.map(e => new Date(e.timestamp).getTime())
  const start = new Date(Math.min(...times))
  const end = new Date(Math.max(...times, Date.now()))
  return { start, end, span: end.getTime() - start.getTime() || 1 }
})

// Pixel scale: 1 day = 80px
const PX_PER_DAY = 80
const totalWidth = computed(() => {
  const days = (timeRange.value.end.getTime() - timeRange.value.start.getTime()) / (24 * 60 * 60 * 1000)
  return Math.max(days * PX_PER_DAY + 200, 800) // min 800px
})

// Position an event on the x-axis
function eventX(timestamp: string): number {
  const t = new Date(timestamp).getTime()
  const dayOffset = (t - timeRange.value.start.getTime()) / (24 * 60 * 60 * 1000)
  return dayOffset * PX_PER_DAY + 100 // 100px left margin
}

// Today marker x position
const todayX = computed(() => eventX(new Date().toISOString()))

// Stagger cards vertically to avoid overlap
// Assign y positions: alternate between two rows (above/below axis)
function eventY(index: number): number {
  // Axis is at y=60. Cards go below.
  // Stagger: alternate 90px and 200px from top
  return index % 2 === 0 ? 90 : 210
}
```

**2. Time axis labels:**

Generate month/day markers along the axis:

```typescript
const axisLabels = computed(() => {
  const labels: { x: number; text: string }[] = []
  const start = new Date(timeRange.value.start)
  start.setHours(0, 0, 0, 0)
  const end = timeRange.value.end

  const current = new Date(start)
  while (current <= end) {
    labels.push({
      x: eventX(current.toISOString()),
      text: current.toLocaleDateString('en-US', { month: 'short', day: 'numeric' }),
    })
    current.setDate(current.getDate() + 7) // weekly labels
  }
  return labels
})
```

**3. Format timestamp for display:**

```typescript
function formatTime(iso: string): string {
  const d = new Date(iso)
  return d.toLocaleDateString('en-US', { month: 'short', day: 'numeric' }) +
    ', ' + d.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit' })
}
```

Following the exact event types and colors from the UI-SPEC §Timeline Event Types:

```typescript
const timelineEvents = computed<TimelineEvent[]>(() => {
  const events: TimelineEvent[] = []

  // Task events
  for (const task of taskStore.tasks) {
    // Task created
    if (task.createdAt) {
      events.push({
        id: `task-${task.id}-created`,
        entityType: 'task',
        entityId: task.id,
        entityTitle: task.title,
        eventType: 'created',
        timestamp: task.createdAt,
        color: '#3B82F6',   // blue-500
        label: 'Task created',
      })
    }
    // Task completed
    if (task.completedAt) {
      events.push({
        id: `task-${task.id}-completed`,
        entityType: 'task',
        entityId: task.id,
        entityTitle: task.title,
        eventType: 'completed',
        timestamp: task.completedAt,
        color: '#10B981',   // emerald-500
        label: 'Task completed',
      })
    }
  }

  // PR events
  for (const pr of [...prStore.myPRs, ...prStore.reviewPRs]) {
    // Dedup by prNumber
    if (events.some(e => e.id === `pr-${pr.prNumber}-opened`)) continue

    // PR opened
    if (pr.createdAt) {
      events.push({
        id: `pr-${pr.prNumber}-opened`,
        entityType: 'pr',
        entityId: pr.prNumber,
        entityTitle: pr.title,
        eventType: 'opened',
        timestamp: pr.createdAt,
        color: '#10B981',   // emerald-500
        label: 'PR opened',
      })
    }
    // PR merged
    if (pr.mergedAt) {
      events.push({
        id: `pr-${pr.prNumber}-merged`,
        entityType: 'pr',
        entityId: pr.prNumber,
        entityTitle: pr.title,
        eventType: 'merged',
        timestamp: pr.mergedAt,
        color: '#8B5CF6',   // violet-500
        label: 'PR merged',
      })
    }
  }

  // Pipeline events
  for (const pipe of adoStore.pipelines) {
    if (pipe.finishTime) {
      const isFailed = pipe.result === 'failed'
      events.push({
        id: `pipeline-${pipe.id}-${pipe.result}`,
        entityType: 'pipeline',
        entityId: pipe.id,
        entityTitle: pipe.name,
        eventType: isFailed ? 'failed' : 'succeeded',
        timestamp: pipe.finishTime,
        color: isFailed ? '#EF4444' : '#22C55E',  // red-500 or green-500
        label: isFailed ? 'Build failed' : 'Build passed',
      })
    }
  }

  // ADO work item events (resolved/closed)
  for (const wi of adoStore.workItems) {
    if (wi.state === 'Resolved' || wi.state === 'Closed') {
      events.push({
        id: `ado-${wi.id}-resolved`,
        entityType: 'ado-work-item',
        entityId: wi.id,
        entityTitle: wi.title,
        eventType: 'resolved',
        timestamp: wi.changedDate || wi.syncedAt,
        color: '#F59E0B',   // amber-500
        label: 'Work item resolved',
      })
    }
  }

  return events.sort((a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime())
})
```

**3. Add navigation handler:**

```typescript
async function handleTimelineNavigate(event: TimelineEvent) {
  switch (event.entityType) {
    case 'task':
      router.push(`/tasks?selected=${event.entityId}`)
      break
    case 'pr': {
      const pr = [...prStore.myPRs, ...prStore.reviewPRs].find(p => p.prNumber === event.entityId)
      if (pr?.prUrl) {
        const { openURL } = await import('@/api/browser')
        openURL(pr.prUrl)
      }
      break
    }
    case 'pipeline': {
      const pipe = adoStore.pipelines.find(p => p.id === event.entityId)
      if (pipe?.url) {
        const { openURL } = await import('@/api/browser')
        openURL(pipe.url)
      }
      break
    }
    case 'ado-work-item': {
      const wi = adoStore.workItems.find(w => w.id === event.entityId)
      if (wi?.url) {
        const { openURL } = await import('@/api/browser')
        openURL(wi.url)
      }
      break
    }
  }
}
```
