package db

import "time"

// PRSnoozeRecord represents a snoozed PR.
type PRSnoozeRecord struct {
    PRID       int       `json:"prId"`
    Repo       string    `json:"repo"`
    SnoozeUntil time.Time `json:"snoozeUntil"`
}

// SnoozePR sets or replaces the snooze time for a PR.
func (db *DB) SnoozePR(prID int, repo string, until time.Time) error {
    _, err := db.Exec(`
        INSERT INTO pr_snooze (pr_id, repo, snooze_until)
        VALUES (?, ?, ?)
        ON CONFLICT(pr_id, repo) DO UPDATE SET
            snooze_until = excluded.snooze_until`,
        prID, repo, until,
    )
    return err
}

// UnSnoozePR removes the snooze for a PR.
func (db *DB) UnSnoozePR(prID int, repo string) error {
    _, err := db.Exec(`DELETE FROM pr_snooze WHERE pr_id = ? AND repo = ?`, prID, repo)
    return err
}

// ListSnoozedPRs returns all currently active snoozes (snooze_until > now).
func (db *DB) ListSnoozedPRs() ([]PRSnoozeRecord, error) {
    rows, err := db.Query(`
        SELECT pr_id, repo, snooze_until FROM pr_snooze
        WHERE snooze_until > CURRENT_TIMESTAMP`)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    var snoozes []PRSnoozeRecord
    for rows.Next() {
        var s PRSnoozeRecord
        if err := rows.Scan(&s.PRID, &s.Repo, &s.SnoozeUntil); err != nil {
            return nil, err
        }
        snoozes = append(snoozes, s)
    }
    return snoozes, rows.Err()
}

// ClearExpiredSnoozes removes snoozes that have passed their expiry.
func (db *DB) ClearExpiredSnoozes() error {
    _, err := db.Exec(`DELETE FROM pr_snooze WHERE snooze_until <= CURRENT_TIMESTAMP`)
    return err
}

// ABDismissal represents a dismissed AB# suggestion.
type ABDismissal struct {
    PRID  int    `json:"prId"`
    AdoID string `json:"adoId"`
}

// DismissABSuggestion records that the user skipped an AB# link suggestion.
func (db *DB) DismissABSuggestion(prID int, adoID string) error {
    _, err := db.Exec(`
        INSERT OR IGNORE INTO ab_dismissals (pr_id, ado_id) VALUES (?, ?)`,
        prID, adoID,
    )
    return err
}

// ListABDismissals returns all dismissed AB# suggestions.
func (db *DB) ListABDismissals() ([]ABDismissal, error) {
    rows, err := db.Query(`SELECT pr_id, ado_id FROM ab_dismissals`)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    var dismissals []ABDismissal
    for rows.Next() {
        var d ABDismissal
        if err := rows.Scan(&d.PRID, &d.AdoID); err != nil {
            return nil, err
        }
        dismissals = append(dismissals, d)
    }
    return dismissals, rows.Err()
}

// SnoozePR snoozes a PR for the given duration preset.
// Valid presets: "1h", "4h", "1d", "1w" (per D-04).
func (s *PRService) SnoozePR(prID int, repo string, preset string) error {
    durations := map[string]time.Duration{
        "1h": time.Hour,
        "4h": 4 * time.Hour,
        "1d": 24 * time.Hour,
        "1w": 7 * 24 * time.Hour,
    }
    d, ok := durations[preset]
    if !ok {
        return fmt.Errorf("invalid snooze preset: %s (valid: 1h, 4h, 1d, 1w)", preset)
    }
    until := time.Now().Add(d)
    return s.db.SnoozePR(prID, repo, until)
}

// UnSnoozePR removes the snooze for a PR.
func (s *PRService) UnSnoozePR(prID int, repo string) error {
    return s.db.UnSnoozePR(prID, repo)
}

// ListSnoozedPRs returns currently active snoozes.
func (s *PRService) ListSnoozedPRs() ([]db.PRSnoozeRecord, error) {
    return s.db.ListSnoozedPRs()
}

// ClearExpiredSnoozes removes expired snooze entries.

// DetectWorkItemRefs scans a PR title and description for AB#12345 patterns.
// Returns unique ADO work item IDs found. Only scans title + first 3 lines of description
// to avoid false positives in code blocks (per pitfall 5 in RESEARCH.md).
func (s *PRService) DetectWorkItemRefs(title string, description string) []int {
    pattern := regexp.MustCompile(`AB#(\d+)`)
    // Limit description to first 3 lines to avoid code block false positives
    descLines := strings.SplitN(description, "\n", 4)
    if len(descLines) > 3 {
        descLines = descLines[:3]
    }
    text := title + "\n" + strings.Join(descLines, "\n")
    matches := pattern.FindAllStringSubmatch(text, -1)
    seen := make(map[int]bool)
    var ids []int
    for _, m := range matches {
        if id, err := strconv.Atoi(m[1]); err == nil && !seen[id] {
            seen[id] = true
            ids = append(ids, id)
        }
    }
    return ids
}

// DismissABSuggestion records that user skipped linking PR to ADO item.
func (s *PRService) DismissABSuggestion(prID int, adoID string) error {
    return s.db.DismissABSuggestion(prID, adoID)
}

// ListABDismissals returns all dismissed AB# suggestions.
func (s *PRService) ListABDismissals() ([]db.ABDismissal, error) {
    return s.db.ListABDismissals()
}
