```go
authService := auth.NewAuthService(database, wailsApp)
```

Register all 4 in `Services: []application.Service{...}`

3. **Add system tray:**
```go
tray := wailsApp.SystemTray.New()
tray.SetMenu(application.NewMenu().
    Add(application.NewMenuItem("Show").OnClick(func(_ *application.Context) {
        wailsApp.Window.Show()
    })).
    AddSeparator().
    Add(application.NewMenuItem("Quit").OnClick(func(_ *application.Context) {
        wailsApp.Quit()
    })),
)
```

4. **Intercept window close to hide:**
```go
wailsApp.Window.RegisterHook(events.Common.WindowClosing, func(e *application.WindowEvent) {
    e.Cancel()
    wailsApp.Window.Hide()
})
```

5. **Attempt silent auth on startup:**
After `wailsApp.Run()` call is set up (or in an OnStartup hook), call:
```go
go func() {
    user, err := authService.TryRestoreSession()
    if err != nil {
        log.Printf("session restore failed: %v", err)
    }
    if user != nil {
        log.Printf("session restored for %s", user.DisplayName)
    }
}()
```

6. **Add import for auth package:**
```go

```

And import for events:


**Step 2: Create router — `frontend/src/router/index.ts`**

```typescript
import { createRouter, createWebHashHistory } from 'vue-router'

const routes = [
  { path: '/', name: 'dashboard', component: () => import('../views/DashboardView.vue') },
  { path: '/tasks', name: 'tasks', component: () => import('../views/TasksView.vue') },
  { path: '/projects', name: 'projects', component: () => import('../views/ProjectsView.vue') },

  history: createWebHashHistory(),
  routes,
})
```

Use hash history (required for Wails — no real server for history mode).

**Step 3: Create Pinia store — `frontend/src/stores/auth.ts`**
```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'

export const useAuthStore = defineStore('auth', () => {
  const user = ref<{ id: string; displayName: string; email: string; avatarUrl: string } | null>(null)
  const isAuthenticated = ref(false)

  function setUser(u: typeof user.value) {
    user.value = u
    isAuthenticated.value = u !== null
  }

  function clearUser() {
    user.value = null
    isAuthenticated.value = false
  }

  return { user, isAuthenticated, setUser, clearUser }
})
```
