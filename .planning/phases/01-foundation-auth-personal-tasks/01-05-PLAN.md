```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

// Wails generates bindings — import from the generated path
// These will be available after first Wails build
// For now, use window.go or dynamic imports

interface Task {
  id: number
  title: string
  description: string
  status: string
  priority: string
  category: string
  projectId: number | null
  area: string
  dueDate: string
  adoId: string
  tags: string
  blockedReason: string
  blockedBy: string
  parentId: number | null
  personalPriority: string
  createdAt: string
  updatedAt: string
  completedAt: string | null
}

export const useTaskStore = defineStore('tasks', () => {
  const tasks = ref<Task[]>([])
  const selectedTask = ref<Task | null>(null)
  const statusFilter = ref('')
  const loading = ref(false)

  // Computed
  const tasksByStatus = computed(() => {
    if (!statusFilter.value) return tasks.value
    return tasks.value.filter(t => t.status === statusFilter.value)
  })

  const stats = computed(() => ({
    total: tasks.value.length,
    todo: tasks.value.filter(t => t.status === 'todo').length,
    inProgress: tasks.value.filter(t => t.status === 'in_progress').length,
    done: tasks.value.filter(t => t.status === 'done').length,
    blocked: tasks.value.filter(t => t.status === 'blocked').length,
  }))

  // Actions — call Go services via Wails bindings
  async function fetchTasks(status = '') {
    loading.value = true
    try {
      // Wails v3 bindings: window.go.app.TaskService.List(status)
      tasks.value = await (window as any).go.app.TaskService.List(status)
    } finally {
      loading.value = false
    }
  }

  async function createTask(title: string, description: string, priority: string, category: string, projectID: number | null, parentID: number | null) {
    const task = await (window as any).go.app.TaskService.Create(title, description, priority, category, projectID || null, parentID || null)
    tasks.value.unshift(task)
    return task
  }

  async function updateTask(id: number, title: string, description: string, status: string, priority: string, category: string, area: string, dueDate: string, tags: string) {
    const task = await (window as any).go.app.TaskService.Update(id, title, description, status, priority, category, area, dueDate, tags)
    const idx = tasks.value.findIndex(t => t.id === id)
    if (idx >= 0) tasks.value[idx] = task
    if (selectedTask.value?.id === id) selectedTask.value = task
    return task
  }

  async function deleteTask(id: number) {
    await (window as any).go.app.TaskService.Delete(id)
    tasks.value = tasks.value.filter(t => t.id !== id)
    if (selectedTask.value?.id === id) selectedTask.value = null
  }

  async function setStatus(id: number, status: string) {
    const task = await (window as any).go.app.TaskService.SetStatus(id, status)
    const idx = tasks.value.findIndex(t => t.id === id)
    if (idx >= 0) tasks.value[idx] = task
    if (selectedTask.value?.id === id) selectedTask.value = task
  }

  async function setPersonalPriority(id: number, priority: string) {
    const task = await (window as any).go.app.TaskService.SetPersonalPriority(id, priority)
    const idx = tasks.value.findIndex(t => t.id === id)
    if (idx >= 0) tasks.value[idx] = task
    if (selectedTask.value?.id === id) selectedTask.value = task
  }

  async function getSubtasks(parentID: number): Promise<Task[]> {
    return await (window as any).go.app.TaskService.GetSubtasks(parentID)
  }

  async function getDependencies(taskID: number): Promise<Task[]> {
    return await (window as any).go.app.DependencyService.GetDependencies(taskID)
  }

  async function addDependency(taskID: number, dependsOn: number) {
    await (window as any).go.app.DependencyService.AddDependency(taskID, dependsOn)
  }

  async function removeDependency(taskID: number, dependsOn: number) {
    await (window as any).go.app.DependencyService.RemoveDependency(taskID, dependsOn)
  }

  async function getAllTags(): Promise<string[]> {
    return await (window as any).go.app.TaskService.GetAllTags()
  }

  return {
    tasks, selectedTask, statusFilter, loading,
    tasksByStatus, stats,
    fetchTasks, createTask, updateTask, deleteTask,
    setStatus, setPersonalPriority,
    getSubtasks, getDependencies, addDependency, removeDependency, getAllTags,
  }
})
```
