const treeRoots = computed(() => {
    let items = workItemTree.value
    // Apply filters
    if (hideLinked.value) items = items.filter(i => !isLinked(i.adoId))
    if (searchQuery.value) {
        const q = searchQuery.value.toLowerCase()
        items = items.filter(i => i.title.toLowerCase().includes(q) || i.adoId.includes(q))
    }
    if (filterType.value !== 'all') items = items.filter(i => i.type === filterType.value)
    if (filterState.value !== 'all') items = items.filter(i => i.state === filterState.value)
    if (filterArea.value !== 'all') items = items.filter(i => i.areaPath.startsWith(filterArea.value))
    // Build tree: items with no parentId or parentId not in the list are roots
    const idSet = new Set(items.map(i => i.id))
    return items.filter(i => !i.parentId || !idSet.has(i.parentId))
})

function getChildren(parentId: number): ADOWorkItem[] {
    return workItemTree.value.filter(i => i.parentId === parentId)
}

Read `internal/app/adoservice.go` and add two methods:

```go
// SavedQuery represents an ADO saved WIQL query
type SavedQuery struct {
    ID       string `json:"id"`
    Name     string `json:"name"`
    Path     string `json:"path"`
    IsFolder bool   `json:"isFolder"`
}

func (s *ADOService) GetSavedQueries() ([]SavedQuery, error) {
    // For each org/project client, call:
    // GET https://dev.azure.com/{org}/{project}/_apis/wit/queries?$depth=1&api-version=7.1
    // Filter to non-folder queries (isFolder=false), return flat list
    // Each query has: id, name, path, isFolder
    clients, err := s.getClients()
    if err != nil { return nil, err }
    var all []SavedQuery
    for _, c := range clients {
        queries, err := ado.GetSavedQueries(c)
        if err != nil {
            log.Printf("[ado] get saved queries %s/%s: %v", c.Org(), c.Project(), err)
            continue
        }
        for _, q := range queries {
            all = append(all, SavedQuery{ID: q.ID, Name: q.Name, Path: q.Path, IsFolder: q.IsFolder})
        }
    }
    return all, nil
}

func (s *ADOService) RunSavedQuery(queryID string) ([]domain.ADOWorkItem, error) {
    // GET https://dev.azure.com/{org}/{project}/_apis/wit/wiql/{id}?api-version=7.1
    // Returns work item IDs, then batch-fetch via GetWorkItemsByIDs
    // Try each client until one succeeds (query belongs to a specific org/project)
    clients, err := s.getClients()
    if err != nil { return nil, err }
    for _, c := range clients {
        items, err := ado.RunSavedQuery(c, queryID)
        if err != nil { continue }
        var result []domain.ADOWorkItem
        for _, item := range items {
            result = append(result, adoWorkItemToDomain(item))
        }
        return result, nil
    }
    return nil, fmt.Errorf("query %s not found in any configured org/project", queryID)
}
```
